    # table 表格

    用于展示大量数据

    # 基础使用

    <w-table
        :columns="columns"
        :data-source="dataSource"
        :selectItem.sync="selectItem"
        :sorter.sync="sorter"
        @update:sorter="getNewSort"
        :loading="loading"
        :height="400"
        bordered
>
    <template slot-scope="item">
        <w-button @click="ViewCol(item.item)">查看</w-button>
        <w-button @click="deleteCol(item.item)">删除</w-button>
    </template>
</w-table>

    # Attributes

    参数 说明 类型 可选值 默认值
    columns 描述表格头的数组对象 Array name / dataIndex width -
    data-source 展示数据的数组对象 Array x(与columns的dataIndex对应) / id / description(添加展开信息) -
    bordered 表格是否有边框 Boolean true / false -
    compacted 表格是否区分单号表格 Boolean true / false -
    select-item 选择表格的item Array - -
    sorter 表格的item的排序 Object x(与columns的dataIndex对应): ""(开启排序) / "asc"(升序) / "desc"(降序) -
    loading table是否展示loading状态 Boolean true / false -
    height table高度 Number - -
    check-able table 表格是否有checkbox Boolean - -
