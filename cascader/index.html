    # cascader 级联选择器

    # 基础使用

    <w-cascader :source="source" height="200px" :selected.sync="selectedCascader" :sync="false" ></w-cascader>

    需要展示的数据结构必须是这样的

    [
    {
        name:"广东",
        children:[{name:"广州"}]
    }
]

    # 异步加载展示数据

    <template>
    <div class="cascader-wrapper">
        <div class="example">
            <w-cascader :source.sync="cascaderData" height="200px" :selected.sync="selectedCascader" :load-data="loadData" :sync="true"></w-cascader>
        </div>
    </div>
</template>

<script>
    import Cascader from "../../../src/cascader/cascader"
    // 在实际项目中, 数据从接口获取
    import dbArray from "../../../tests/fixtures/db"

    // 1. 编写函数ajax，输入id，输出一个数组对象，对象的parent_id与输入id一致
    function ajax(parent_id = 0){
        return new Promise((resolve, reject)=>{
            let id = setTimeout(()=>{
                let result = dbArray.filter((obj)=>{
                    if(obj.parent_id === parent_id){
                        return obj
                    }
                })
                result.forEach((obj)=>{
                    const childrenNode = dbArray.filter((item)=>{
                        if(item.parent_id === obj.id){
                            return item
                        }
                    })
                    obj.isLeaf = childrenNode.length <= 0;
                })


                resolve(result)
            }, 1000)
            return id
        })
    }
    export default {
        components: {
            "w-cascader": Cascader,
        },
        data() {
            return {
                cascaderData: [],
                selectedCascader: [],
            }
        },
        methods:{
            // 2. 编写 loadData 函数，该函数是组件提供的回调函数
            loadData(node, fn){
                const {id}  = node;
                ajax(id).then((result)=>{
                    fn(result)
                })
            },
        },
        // 3. 合适时机初始化第一层展示数据
        created(){
            ajax().then((resolve)=>{
                this.cascaderData = resolve;
            })
        }
    }
</script>
